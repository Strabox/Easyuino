39  /*
40  GSMServiceSecure expands the GSMService API allowing user to add phone numbers
41  to filter calls and SMSs.
42  @Physical Devices Supported:
43  - SIM90x Series Chipset, SIM80x Chipset
44  @Physical Devices Tested:
45  - SIM900 Chipset
46  */
47  class GSMServiceSecure : public GSMService {
48 
49  private:
50  /* The numbers that are allowed to send SMS to the module */
51  unsigned long _allowedNumbers[MAX_NUMBER_OF_ALLOWED_NUMBERS];
52 
53  public:
54  /*
55  Creates a GSMServiceSecure API object.
56  @param txPin - Pin used to send the serial information to GSM module
57  @param rxPin - Pin used to receive the serial information from GSM module
58  @param powerPin - Pin used to turn on and off the GSM module
59  @param outputStream - Stream to output the GSM Module outputs in case of necessary debug
60  */
61  GSMServiceSecure(IN uint8_t txPin, IN uint8_t rxPin, IN uint8_t powerPin, IN Stream &outputStream);
62 
63  /*
64  Creates a GSMServiceSecure API object.
65  @param txPin - Pin used to send the serial information to GSM module
66  @param rxPin - Pin used to receive the serial information from GSM module
67  @param powerPin - Pin used to turn on and off the GSM module
68  */
69  GSMServiceSecure(IN uint8_t txPin, IN uint8_t rxPin, IN uint8_t powerPin);
70 
71  /* Deallocates all the internal resources used by the GSMService */
72  ~GSMServiceSecure();
73 
74 #pragma region Controller Numbers Management
75 
76  /*
77  Add the phone number to the allowed numbers IF it doesn't exist yet.
78  @param phoneNumberToAdd - Phone number to add to the allowed numbers
79  @return - GSMService API request status (NOT_INITIALIZED, GSM_REQUEST_INVALID_ARGUMENT, GSM_OK, GSM_MAXIMUM_ALLOWED_NUMBERS_REACHED)
80  */
81  GSMRequestStatus addAllowedNumber(IN unsigned long phoneNumberToAdd);
82 
83  /*
84  Verify if the phone number is in the list of allowed numbers
85  @param phoneNumber - Number to verify
86  @param allowed - After the method executes. True: If the number is in allowed list or if the list is empty. False: Otherwise.
87  @return - GSMService API request status (NOT_INITIALIZED, GSM_REQUEST_INVALID_ARGUMENT, GSM_OK)
88  */
89  GSMRequestStatus isAllowed(IN unsigned long phoneNumber, OUT bool& allowed);
90 
91  /*
92  Remove the given number from the allowed numbers list (If exists there)
93  @param phoneNumberToRemove - Number to remove from the allowed numbers
94  @return - GSMService API request status
95  */
96  GSMRequestStatus removeAllowedNumber(IN unsigned long phoneNumberToRemove);
97 
98  /*
99  Clear the list for allowed phone numbers ENABLING any number to Call and send SMS to the service and accept it.
100  @return - GSMService API request status (NOT_INITIALIZED, GSM_OK)
101  */
102  GSMRequestStatus clearAllowedNumbers();
103 
104 #pragma endregion
105

108  virtual GSMRequestStatus availableSMS(INOUT SMS &message, OUT bool &smsRead);

110 #pragma endregion
